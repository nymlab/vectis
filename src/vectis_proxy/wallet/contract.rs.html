use cosmwasm_std::{from_binary, Addr, Binary, Order, Response, StdError, StdResult};
use sylvia::types::{ExecCtx, QueryCtx};
use sylvia_derive::contract;
use vectis_wallet::{
    interface::{
        factory_management_trait::sv::{Querier, Remote as FactoryRemote},
        wallet_trait, WalletTrait,
    },
    types::{
        error::WalletError,
        factory::ChainConnection,
        state::VectisActors,
        wallet::{Controller, RelayTransaction, WalletAddrs, WalletInfo},
    },
    util::query::{get_ica, get_items_from_deployer},
    POLICY,
};

use crate::contract::VectisProxy;
use crate::wallet::exec;

#[contract(module=crate::contract)]
#[messages(wallet_trait as WalletTrait)]
impl<'a> WalletTrait for VectisProxy<'a> {
    type Error = WalletError;

    /// Returns the wallet info
    #[msg(query)]
    fn info(&self, ctx: QueryCtx) -> Result<WalletInfo, StdError> {
        let QueryCtx { deps, env } = ctx;
        let contract_info = deps
            .querier
            .query_wasm_contract_info(env.contract.address.clone())?;

        let factory = get_items_from_deployer(deps, VectisActors::Factory)
            .map_err(|e| StdError::generic_err(e.to_string()))?;

        // We are using query wasm smart here as we cannot directly use SUPPORTEDCHAINS.query(...)
        // as we do not know what key to query.
        let supported_chains = FactoryRemote::new(Addr::unchecked(factory))
            .querier(&deps.querier)
            .supported_chains(None, None)
            .map_err(|e| StdError::generic_err(e.to_string()))?;

        let mut addresses = vec![];
        for chain in supported_chains {
            match chain.1 {
                ChainConnection::IBC(connection_id) => {
                    addresses.push(WalletAddrs {
                        chain_id: chain.0,
                        addr: get_ica(deps, env.contract.address.to_string(), connection_id).ok(),
                    });
                }
                ChainConnection::Other(_) => {
                    addresses.push(WalletAddrs {
                        chain_id: chain.0.clone(),
                        addr: self
                            .data
                            .may_load(deps.storage, chain.0.as_bytes())?
                            .map(|b| -> Result<String, _> { from_binary(&b) })
                            .transpose()?,
                    });
                }
            }
        }

        Ok(WalletInfo {
            controller: self.controller.load(deps.storage)?,
            deployer: deps.api.addr_humanize(&self.deployer.load(deps.storage)?)?,
            version: cw2::get_contract_version(deps.storage)?,
            code_id: contract_info.code_id,
            relayers: self
                .relayers
                .keys(deps.storage, None, None, Order::Ascending)
                .collect::<StdResult<Vec<Addr>>>()?,
            created_at: self.created_at.load(deps.storage)?,
            vid: self.vid.load(deps.storage)?,
            addresses,
            policy: self.data.may_load(deps.storage, POLICY)?,
        })
    }

    /// Returns the data given the key
    #[msg(query)]
    fn data(&self, ctx: QueryCtx, key: Binary) -> Result<Option<Binary>, StdError> {
        self.data.may_load(ctx.deps.storage, &key)
    }

    #[msg(exec)]
    fn controller_nonce_update(&self, ctx: ExecCtx) -> Result<Response, Self::Error> {
        let factory = get_items_from_deployer(ctx.deps.as_ref(), VectisActors::Factory)?;
        if ctx.info.sender != Addr::unchecked(factory) {
            return Err(WalletError::InvalidAuthenticator);
        }
        self.controller
            .update(ctx.deps.storage, |mut controller| -> StdResult<_> {
                controller.increment_nonce();
                Ok(controller)
            })?;

        Ok(Response::default())
    }

    #[msg(exec)]
    fn auth_exec(
        &self,
        ctx: ExecCtx,
        transaction: RelayTransaction,
    ) -> Result<Response, Self::Error> {
        let ExecCtx { deps, env, .. } = ctx;
        let controller = self.controller.load(deps.storage)?;

        let msgs = exec::auth_check(self, deps.as_ref(), transaction, controller)?;
        self.controller
            .update(deps.storage, |mut controller| -> StdResult<_> {
                controller.increment_nonce();
                Ok(controller)
            })?;

        if self.pre_tx_plugins.is_empty(deps.storage)
            && self.post_tx_hooks_plugins.is_empty(deps.storage)
        {
            Ok(Response::new().add_messages(msgs))
        } else {
            self.subscription_check(deps.as_ref(), &env)?;
            self.pre_tx_check(deps.as_ref(), &msgs)?;
            let post_tx_hook_msgs = self.exec_with_post_tx_hooks(deps, msgs.clone())?;
            Ok(Response::new()
                .add_messages(msgs)
                .add_messages(post_tx_hook_msgs))
        }
    }

    #[msg(exec)]
    fn auth_exec_without_plugins(
        &self,
        ctx: ExecCtx,
        transaction: RelayTransaction,
    ) -> Result<Response, Self::Error> {
        let ExecCtx { deps, .. } = ctx;
        let controller = self.controller.load(deps.storage)?;

        let msgs = exec::auth_check(self, deps.as_ref(), transaction, controller)?;

        self.controller
            .update(deps.storage, |mut controller| -> StdResult<_> {
                controller.increment_nonce();
                Ok(controller)
            })?;

        Ok(Response::new().add_messages(msgs))
    }

    #[msg(exec)]
    fn controller_rotation(
        &self,
        ctx: ExecCtx,
        new_controller: Controller,
    ) -> Result<Response, Self::Error> {
        // This allows self to rotate / guardian plugin to do so
        self.ensure_is_contract_self(&ctx.env, &ctx.info.sender)?;
        exec::rotation(self, ctx.deps, new_controller)
    }

    #[msg(exec)]
    fn update_data(
        &self,
        ctx: ExecCtx,
        data: Vec<(Binary, Option<Binary>)>,
    ) -> Result<Response, Self::Error> {
        self.ensure_is_contract_self(&ctx.env, &ctx.info.sender)?;
        exec::update_data(self, ctx.deps, data)
    }
}
