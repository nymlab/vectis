use super::{config, plugin_mgmt, query};
use crate::contract::VectisPluginRegistry;
use cw2::{get_contract_version, ContractVersion};
use sylvia::types::{ExecCtx, QueryCtx};
use sylvia_derive::contract;
use vectis_wallet::{
    interface::{registry_management_trait, RegistryManagementTrait},
    types::{
        error::PluginRegError,
        plugin::{
            Plugin, PluginCodeData, PluginMetadataData, PluginWithVersionResponse, PluginsResponse,
        },
        plugin_registry::{RegistryConfigResponse, TierDetails},
    },
};

use cosmwasm_std::{Coin, Response, StdError, StdResult};

#[contract(module=crate::contract)]
#[messages(registry_management_trait as RegistryManagementTrait)]
impl<'a> RegistryManagementTrait for VectisPluginRegistry<'a> {
    type Error = PluginRegError;
    #[msg(exec)]
    fn register_plugin(
        &self,
        ctx: ExecCtx,
        code_data: PluginCodeData,
        metadata_data: PluginMetadataData,
    ) -> Result<Response, Self::Error> {
        self.ensure_is_reviewer(&ctx)?;
        plugin_mgmt::register(self, ctx.deps, ctx.info, code_data, metadata_data)
    }

    #[msg(exec)]
    fn unregister_plugin(&self, ctx: ExecCtx, id: u64) -> Result<Response, Self::Error> {
        self.ensure_is_reviewer(&ctx)?;
        plugin_mgmt::unregister(self, ctx.deps, id)
    }

    #[msg(exec)]
    fn new_plugin_version(
        &self,
        ctx: ExecCtx,
        /// The id on the vectis plugin registry
        id: u64,
        /// Code update must pump latest_contract_version
        code_update: Option<PluginCodeData>,
        /// Metadata update will not require code version pump
        metadata_update: PluginMetadataData,
    ) -> Result<Response, Self::Error> {
        self.ensure_is_reviewer(&ctx)?;
        plugin_mgmt::new_plugin_version(self, ctx.deps, id, code_update, metadata_update)
    }

    #[msg(exec)]
    fn update_registry_fee(&self, ctx: ExecCtx, new_fee: Coin) -> Result<Response, Self::Error> {
        self.ensure_is_reviewer(&ctx)?;
        config::update_registry_fee(self, ctx.deps, new_fee)
    }

    /// Adding new tiers
    /// To remove tier there may already be subscribers and so it will require a migration function
    #[msg(exec)]
    fn add_or_update_subscription_tiers(
        &self,
        ctx: ExecCtx,
        tier: u8,
        details: TierDetails,
    ) -> Result<Response, Self::Error> {
        self.ensure_is_reviewer(&ctx)?;
        config::add_or_update_subscription_tiers(self, ctx.deps, tier, details)
    }

    #[msg(query)]
    fn get_plugins(
        &self,
        ctx: QueryCtx,
        limit: Option<u32>,
        start_after: Option<u32>,
    ) -> StdResult<PluginsResponse> {
        query::plugins(self, ctx.deps, limit, start_after)
    }

    #[msg(query)]
    fn get_plugin_by_id(&self, ctx: QueryCtx, id: u64) -> StdResult<Option<Plugin>> {
        self.plugins.may_load(ctx.deps.storage, id)
    }

    /// This helps to do all the neccessary queries to allow caller to know the ipfs_hash
    #[msg(query)]
    fn get_plugin_by_address(
        &self,
        ctx: QueryCtx,
        contract_addr: String,
    ) -> StdResult<PluginWithVersionResponse> {
        query::plugin_by_address(self, ctx.deps, contract_addr)
    }

    #[msg(query)]
    fn get_config(&self, ctx: QueryCtx) -> StdResult<RegistryConfigResponse> {
        query::config(self, ctx.deps)
    }

    #[msg(query)]
    fn contract_version(&self, ctx: QueryCtx) -> Result<ContractVersion, StdError> {
        get_contract_version(ctx.deps.storage)
    }
}
