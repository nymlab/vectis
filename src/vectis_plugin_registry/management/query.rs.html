use cosmwasm_std::{Deps, Order, StdResult};

use crate::contract::VectisPluginRegistry;

use vectis_wallet::{
    types::{
        plugin::{Plugin, PluginWithVersionResponse, PluginsResponse},
        plugin_registry::{RegistryConfigResponse, TierDetails},
    },
    DEFAULT_LIMIT, MAX_LIMIT,
};

pub(crate) fn plugins(
    c: &VectisPluginRegistry,
    deps: Deps,
    limit: Option<u32>,
    start_after: Option<u32>,
) -> StdResult<PluginsResponse> {
    let limit = limit.unwrap_or(DEFAULT_LIMIT).min(MAX_LIMIT) as usize;
    let start = start_after.map(cw_storage_plus::Bound::exclusive);

    let plugins = c
        .plugins
        .range(deps.storage, start, None, Order::Ascending)
        .take(limit)
        .map(|item| item.map(|(_, v)| v))
        .collect::<StdResult<Vec<Plugin>>>()?;

    Ok(PluginsResponse {
        plugins,
        total: c.total_plugins.load(deps.storage)?,
        current_plugin_id: c.current_plugin_id.load(deps.storage)?,
    })
}

/// This helps to do all the neccessary queries to allow called to know the ipfs_hash
pub(crate) fn plugin_by_address(
    c: &VectisPluginRegistry,
    deps: Deps,
    contract_addr: String,
) -> StdResult<PluginWithVersionResponse> {
    let addr = deps.api.addr_validate(&contract_addr)?;

    let code_info = deps.querier.query_wasm_contract_info(addr.as_str())?;
    let contract_info = c.plugin_contract_state.query(&deps.querier, addr)?;

    let plugin_id = c.plugin_code_ids.load(deps.storage, code_info.code_id)?;

    let plugin = c.plugins.load(deps.storage, plugin_id)?;

    Ok(PluginWithVersionResponse {
        contract_version: contract_info.version,
        plugin_info: plugin,
    })
}

pub(crate) fn config(c: &VectisPluginRegistry, deps: Deps) -> StdResult<RegistryConfigResponse> {
    Ok(RegistryConfigResponse {
        registry_fee: c.registry_fee.load(deps.storage)?,
        deployer_addr: deps
            .api
            .addr_humanize(&c.deployer.load(deps.storage)?)?
            .to_string(),
        subscription_tiers: c
            .subscription_tiers
            .prefix(())
            .range(deps.storage, None, None, Order::Ascending)
            .collect::<StdResult<Vec<(u8, TierDetails)>>>()?,
    })
}
