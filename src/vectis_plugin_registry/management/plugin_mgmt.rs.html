use cosmwasm_std::{
    ensure_eq, BankMsg, CosmosMsg, DepsMut, MessageInfo, Response, StdResult, Uint128,
};

use crate::contract::VectisPluginRegistry;

use vectis_wallet::types::{
    error::PluginRegError,
    plugin::{Plugin, PluginCodeData, PluginMetadataData, VersionDetails},
};

pub(crate) fn register(
    c: &VectisPluginRegistry,
    deps: DepsMut,
    info: MessageInfo,
    code_data: PluginCodeData,
    metadata_data: PluginMetadataData,
) -> Result<Response, PluginRegError> {
    // Check if the code_id was already plugin_registered
    if c.plugin_code_ids.has(deps.storage, code_data.new_code_id) {
        return Err(PluginRegError::CodeIdAlreadyRegistered);
    }

    // Check if the caller has enough funds to pay the fee
    let registry_fee = c.registry_fee.load(deps.storage)?;
    if registry_fee.amount != Uint128::zero() {
        if info.funds.len() != 1 {
            return Err(PluginRegError::TooManyCoins);
        }
        if info.funds[0] != registry_fee {
            return Err(PluginRegError::InsufficientFee(
                registry_fee.amount,
                info.funds[0].amount,
            ));
        }
    }

    if registry_fee.amount == Uint128::zero() && !info.funds.is_empty() {
        return Err(PluginRegError::IncorrectFees(registry_fee));
    }

    // Store plugin information in PLUGINS Map<u64(id), Plugin>
    let id = c
        .current_plugin_id
        .update(deps.storage, |total| -> StdResult<_> {
            Ok(total.checked_add(1).expect("overflow"))
        })?;
    c.total_plugins
        .update(deps.storage, |total| -> StdResult<_> {
            Ok(total.checked_add(1).expect("overflow"))
        })?;

    // Do checks on the validity of the info provided
    let code_info = deps.querier.query_wasm_code_info(code_data.new_code_id)?;
    ensure_eq!(
        code_info.checksum.to_hex(),
        code_data.new_code_hash,
        PluginRegError::InvalidChecksum
    );

    // Save to Plugins
    let plugin = Plugin::new(
        deps.as_ref(),
        id,
        metadata_data.creator,
        metadata_data.display_name,
        metadata_data.ipfs_hash,
        code_data.latest_contract_version,
        code_data.new_code_id,
        code_data.new_code_hash,
    )?;

    c.plugins.save(deps.storage, id, &plugin)?;
    c.plugin_code_ids
        .save(deps.storage, code_data.new_code_id, &id)?;
    let event = c
        .get_event("plugin_registered")
        .add_attribute("id", id.to_string());

    // Send funds to the Deployer if required
    if registry_fee.amount != Uint128::zero() {
        let msg = CosmosMsg::Bank(BankMsg::Send {
            to_address: deps
                .api
                .addr_humanize(&c.deployer.load(deps.storage)?)?
                .to_string(),
            amount: vec![registry_fee],
        });
        Ok(Response::new().add_event(event).add_message(msg))
    } else {
        Ok(Response::new().add_event(event))
    }
}

pub(crate) fn unregister(
    c: &VectisPluginRegistry,
    deps: DepsMut,
    id: u64,
) -> Result<Response, PluginRegError> {
    let plugin = c
        .plugins
        .may_load(deps.storage, id)?
        .ok_or(PluginRegError::PluginNotFoundOnRegistry(id))?;

    // remove from plugind_code_ids
    for v in plugin.versions.values() {
        c.plugin_code_ids.remove(deps.storage, v.code_id);
    }

    // remove from plugins
    c.plugins.remove(deps.storage, id);

    // deduct one from total
    c.total_plugins
        .update(deps.storage, |total| -> StdResult<_> {
            Ok(total.checked_sub(1).expect("overflow"))
        })?;

    let event = c
        .get_event("plugin_unregistered")
        .add_attribute("id", id.to_string());
    Ok(Response::new().add_event(event))
}

pub(crate) fn new_plugin_version(
    c: &VectisPluginRegistry,
    deps: DepsMut,
    id: u64,
    code_update: Option<PluginCodeData>,
    metadata_update: PluginMetadataData,
) -> Result<Response, PluginRegError> {
    // Checks if there is existing version
    let mut plugin = c.plugins.load(deps.storage, id)?;

    plugin.creator = deps
        .api
        .addr_canonicalize(deps.api.addr_validate(&metadata_update.creator)?.as_str())?;
    plugin.display_name = metadata_update.display_name;

    // If None: there is no code update and purely metadata update
    if let Some(code_data) = code_update {
        // Check if the code_id was already plugin_registered
        if c.plugin_code_ids.has(deps.storage, code_data.new_code_id) {
            return Err(PluginRegError::CodeIdAlreadyRegistered);
        }

        let latest_version = code_data.latest_contract_version;
        // If there is a code update, we check the version has beeen pumped
        if plugin.versions.contains_key(&latest_version) {
            return Err(PluginRegError::VersionExists(latest_version));
        }

        // Check that the code hash provided is correct
        let code_info = deps.querier.query_wasm_code_info(code_data.new_code_id)?;
        ensure_eq!(
            code_info.checksum.to_hex(),
            code_data.new_code_hash,
            PluginRegError::InvalidChecksum
        );

        plugin.latest_contract_version = latest_version.clone();
        plugin.versions.insert(
            latest_version,
            VersionDetails {
                code_id: code_data.new_code_id,
                code_hash: code_data.new_code_hash,
                ipfs_hash: metadata_update.ipfs_hash,
            },
        );
        c.plugin_code_ids
            .save(deps.storage, code_data.new_code_id, &id)?;
    } else {
        // update ipfs hash without code_data
        if let Some(p) = plugin.versions.get_mut(&plugin.latest_contract_version) {
            p.ipfs_hash = metadata_update.ipfs_hash;
        }
    }

    // replace the plugin in storage for the given id
    c.plugins.save(deps.storage, id, &plugin)?;

    let event = c
        .get_event("new_plugin_version")
        .add_attribute("id", id.to_string());
    Ok(Response::new().add_event(event))
}
