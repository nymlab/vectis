use crate::contract::VectisFactory;
use cosmwasm_std::{ensure, ensure_eq, to_binary, CosmosMsg, DepsMut, Response, WasmMsg};
use vectis_wallet::{
    interface::wallet_trait,
    types::{
        error::{FactoryError, MigrationMsgError},
        factory::MigrateWalletMsg,
        state::CONTROLLER,
    },
    util::signature::relay_tx_auth_check,
};

/// Migrates the instantiated `wallet_proxy` instance to a new code id
pub(crate) fn migrate_wallet(
    c: &VectisFactory,
    deps: DepsMut,
    migrations_msg: MigrateWalletMsg,
) -> Result<Response, FactoryError> {
    let wallet = deps.api.addr_validate(&migrations_msg.addr_to_migrate)?;
    let controller = CONTROLLER.query(&deps.querier, wallet.clone())?;

    let auth_addr = c
        .authenticators
        .may_load(deps.storage, controller.auth.ty().to_string())?;
    let mut msgs = relay_tx_auth_check(deps.as_ref(), controller, migrations_msg.tx, auth_addr)?;
    ensure!(msgs.len() == 1, MigrationMsgError::InvalidMigrationMsg);

    let cosmos_msg = msgs.pop().unwrap();
    if let CosmosMsg::Wasm(WasmMsg::Migrate {
        contract_addr,
        new_code_id,
        ..
    }) = &cosmos_msg
    {
        // Ensure user signed to migrate their own wallet
        ensure_eq!(
            contract_addr,
            &migrations_msg.addr_to_migrate,
            MigrationMsgError::InvalidWalletAddr
        );
        // Ensure code_id is supported code_id
        ensure!(
            c.supported_proxies.has(deps.storage, *new_code_id),
            MigrationMsgError::UnsupportedProxyId
        );
    } else {
        return Err(MigrationMsgError::InvalidMigrationMsg.into());
    }

    let update_nonce_msg = CosmosMsg::Wasm(WasmMsg::Execute {
        contract_addr: migrations_msg.addr_to_migrate,
        msg: to_binary(&wallet_trait::sv::ExecMsg::ControllerNonceUpdate {})?,
        funds: vec![],
    });

    let event = c
        .get_event("wallet_migration")
        .add_attribute("addr", wallet);

    Ok(Response::new()
        .add_event(event)
        .add_message(cosmos_msg)
        .add_message(update_nonce_msg))
}
