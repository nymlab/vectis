use cosmwasm_std::{from_binary, Addr, Response, StdError};
use sylvia::types::{ExecCtx, QueryCtx};
use sylvia_derive::contract;
use vectis_wallet::{
    interface::{factory_service_trait, FactoryServiceTrait},
    types::{
        error::FactoryError,
        factory::{ChainConnection, CreateWalletMsg, MigrateWalletMsg},
        state::PROXY_DATA,
    },
    util::query::get_ica,
};

use crate::contract::VectisFactory;
use crate::service::{create, migrate};

#[contract(module=crate::contract)]
#[messages(factory_service_trait as FactoryServiceTrait)]
impl<'a> FactoryServiceTrait for VectisFactory<'a> {
    type Error = FactoryError;

    #[msg(exec)]
    fn create_wallet(
        &self,
        ctx: ExecCtx,
        create_wallet_msg: CreateWalletMsg,
    ) -> Result<Response, Self::Error> {
        let ExecCtx { deps, env, info } = ctx;

        // only authorised persons can create wallet
        let creator = self.wallet_creator.load(deps.storage)?;
        if info.sender != creator {
            return Err(FactoryError::Unauthorized);
        }

        create::create_wallet(self, deps, info, env, create_wallet_msg)
    }

    #[msg(exec)]
    fn migrate_wallet(
        &self,
        ctx: ExecCtx,
        migration_msg: MigrateWalletMsg,
    ) -> Result<Response, Self::Error> {
        migrate::migrate_wallet(self, ctx.deps, migration_msg)
    }

    /// Returns the wallet address of this label
    #[msg(query)]
    fn wallet_by_vid(&self, ctx: QueryCtx, vid: String) -> Result<Option<Addr>, StdError> {
        self.controllers.may_load(ctx.deps.storage, &vid)
    }

    /// Returns the address of the vid on the supported chain
    #[msg(query)]
    fn wallet_by_vid_chain(
        &self,
        ctx: QueryCtx,
        vid: String,
        chain_id: String,
    ) -> Result<Option<String>, StdError> {
        let owner = self.controllers.load(ctx.deps.storage, &vid)?;

        let chain_connection = self.supported_chains.load(ctx.deps.storage, &chain_id)?;

        match chain_connection {
            ChainConnection::IBC(connection_id) => {
                get_ica(ctx.deps, owner.to_string(), connection_id).map(Some)
            }
            ChainConnection::Other(_) => {
                let data = PROXY_DATA.query(&ctx.deps.querier, owner, chain_id.as_bytes())?;
                data.map(|binary| from_binary(&binary)).transpose()
            }
        }
    }
}
