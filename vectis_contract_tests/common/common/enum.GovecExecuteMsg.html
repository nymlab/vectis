pub enum GovecExecuteMsg {
    Transfer {
        recipient: String,
        amount: Uint128,
        relayed_from: Option<String>,
    },
    TransferFrom {
        owner: String,
        recipient: String,
        amount: Uint128,
    },
    Burn {
        amount: Uint128,
    },
    Exit {
        relayed_from: Option<String>,
    },
    Send {
        contract: String,
        amount: Uint128,
        msg: Binary,
        relayed_from: Option<String>,
    },
    Mint {
        new_wallet: String,
    },
    UpdateMintCap {
        new_mint_cap: Option<Uint128>,
    },
    UpdateMintAmount {
        new_amount: Uint128,
    },
    UpdateDaoAddr {
        new_addr: String,
    },
    UpdateMarketing {
        project: Option<String>,
        description: Option<String>,
        marketing: Option<String>,
    },
    UploadLogo(Logo),
}
